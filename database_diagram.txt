- id INT (PK)

- name VARCHAR(255)
- description TEXT
- characteristics JSON
- is_predefined TINYINT(1)
- is_active TINYINT(1)
- created_at TIMESTAMP
- updated_at TIMESTAMP

Table: EvaluationUserAgents
- id INT (PK)
- eval_id INT
- info_prompt TEXT
- user_agent_definition_id INT (FK to UserAgentDefinitions.id)

Table: DocumentationSources
- id INT (PK)
- product_id INT
- type VARCHAR(50)
- url VARCHAR(255)
- created_at TIMESTAMP
- updated_at TIMESTAMP
- storage_url VARCHAR(255)
- file_size BIGINT
- content_type VARCHAR(100)
- fetched_at TIMESTAMP
